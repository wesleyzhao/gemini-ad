# Automated email categorization import gemini # Initialize Gemini client client = gemini.Client(api_key) # Process inbox for email in inbox: category = client.categorize( text=email.body, categories=["urgent", "normal", "spam"] ) email.apply_label(category)

// Batch data processing const gemini = require('@google/gemini'); // Process CSV data async function processData(csvFile) { const data = await gemini.analyze({ file: csvFile, task: 'extract-insights' }); return data.toJSON(); }
