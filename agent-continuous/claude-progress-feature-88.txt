   ```javascript
                      window.location.hostname === '127.0.0.1' ||
                      window.location.search.includes('debug=true');

   if (DEBUG_MODE) console.log('[Web Vital] LCP:', value);
   ```

2. **Lazy Loading Implementation (32 optimizations)**
   - Added loading="lazy" to images across 9 pages
   - Implemented IntersectionObserver fallback
   - Improved page load performance (LCP)
   - Reduced bandwidth consumption

   Implementation:
   ```html
   <img src="image.jpg" alt="Feature" loading="lazy">
   ```

   JavaScript Fallback:
   ```javascript
   if ('IntersectionObserver' in window) {
     const lazyImages = document.querySelectorAll('img[loading="lazy"]');
     const imageObserver = new IntersectionObserver((entries) => {
       entries.forEach(entry => {
         if (entry.isIntersecting) {
           const img = entry.target;
           if (img.dataset.src) {
             img.src = img.dataset.src;
             img.removeAttribute('data-src');
           }
           imageObserver.unobserve(img);
         }
       });
     });
     lazyImages.forEach(img => imageObserver.observe(img));
   }
   ```

3. **IntersectionObserver Setup (9 optimizations)**

   - Generate comprehensive reports

   - Provide optimization recommendations

**Configuration:**
```javascript
const CONFIG = {
  pages: 13,
```

**Usage:**
